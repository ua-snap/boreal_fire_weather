def find_file(dir, fn):
    for root, dirs, files in os.walk(dir):
        if fn in files:
            return os.path.join(root, fn)
    return None

def plot_nan_comparison(var, fn1, fn2, fn3, timestep):
    ds1 = xr.open_dataset(fn1)
    ds2 = xr.open_dataset(fn2)
    ds3 = xr.open_dataset(fn3)

    arr1 = ds1[var].isel(time=timestep)
    arr2 = ds2[var].isel(time=timestep)
    arr3 = ds3[var].isel(time=timestep)

    # get date string for the random timestep
    datestr = str(ds1['time'].isel(time=timestep).values)[:10]
    print(f"Random timestep: {timestep} ({datestr})")

    # Create a mask for NaN values in both datasets
    print(f"Number of NaN values in original dataset: {arr1.isnull().sum().item()}")
    print(f"Number of NaN values in data release dataset: {arr2.isnull().sum().item()}")
    print(f"Number of NaN values in revised dataset: {arr3.isnull().sum().item()}")

    # build colorbar using range of both datasets
    cbar_min = min(arr1.min().item(), arr2.min().item(), arr3.min().item())
    cbar_max = max(arr1.max().item(), arr2.max().item(), arr3.max().item())

    fig, axes = plt.subplots(1, 3, figsize=(18, 6), sharey=True)
    plt.suptitle(f"{var.upper()} Comparison: {fn1.split('/')[-1]} : {datestr}", fontsize=16)

    arr1.plot(ax=axes[0], vmin=cbar_min, vmax=cbar_max, cmap='viridis')
    axes[0].set_title(f'{var.upper()} - Original')

    arr2.plot(ax=axes[1], vmin=cbar_min, vmax=cbar_max, cmap='viridis')
    axes[1].set_title(f'{var.upper()} - Data Release')

    arr3.plot(ax=axes[2], vmin=cbar_min, vmax=cbar_max, cmap='viridis')
    axes[2].set_title(f'{var.upper()} - Revised by JP')

    plt.tight_layout(rect=[0, 0, 1, 0.95])
    plt.show()
    

def run_comparison(fn, var, timestep):

    fn1 = find_file(original_data_dir, fn)
    fn2 = find_file(data_release_dir, fn)
    fn3 = find_file(revised_dir, fn)
    
    print(f"Comparing files:\n{fn1}\n{fn2}\n{fn3}")
    
    plot_nan_comparison(var, fn1, fn2, fn3, timestep)
